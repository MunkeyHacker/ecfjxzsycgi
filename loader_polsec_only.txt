local function trim(v)
    v = tostring(v or "")
    return v:match("^%s*(.-)%s*$") or ""
end

local function parentList()
    local list = {}

    local ok, hui = pcall(function()
        return gethui and gethui() or nil
    end)
    if ok and hui then
        table.insert(list, hui)
    end

    local playerGui = LocalPlayer:FindFirstChildOfClass("PlayerGui") or LocalPlayer:WaitForChild("PlayerGui", 10)
    if playerGui then
        table.insert(list, playerGui)
    end

    local coreGui
    ok, coreGui = pcall(function()
        return game:GetService("CoreGui")
    end)
    if ok and coreGui then
        table.insert(list, coreGui)
    end

    return list
end

local function disconnect(c)
    if c and typeof(c) == "RBXScriptConnection" then
        pcall(function()
            c:Disconnect()
        end)
    end
end

local function cleanup()
    for _, c in ipairs(conns) do
        disconnect(c)
    end
    conns = {}

    pcall(function()
        if guiRef and guiRef.Parent then
            guiRef:Destroy()
        end
    end)
    guiRef = nil

    for _, parent in ipairs(parentList()) do
        pcall(function()
            local names = {
                "CoroutineKeyLoader",
                "CoroutinePolsecKeyPreloader",
                "CoroutinePolsecLoaderStatus",
                "StandPolsecKeyPreloader",
                "StandPolsecLoaderStatus",
            }
            for _, name in ipairs(names) do
                local old = parent:FindFirstChild(name)
                if old then old:Destroy() end
            end
        end)
    end
end

local function track(c)
    if c and typeof(c) == "RBXScriptConnection" then
        table.insert(conns, c)
    end
    return c
end

local function corner(parent, radius)
    local c = Instance.new("UICorner")
    c.CornerRadius = UDim.new(0, radius or 8)
    c.Parent = parent
    return c
end

local function pad(parent, left, right)
    local p = Instance.new("UIPadding")
    p.PaddingLeft = UDim.new(0, left or 0)
    p.PaddingRight = UDim.new(0, right or left or 0)
    p.Parent = parent
    return p
end

local function setProgress(amount, text)
    amount = math.clamp(tonumber(amount) or 0, 0, 1)
    if progressFill then
        progressFill:TweenSize(UDim2.new(amount, 0, 1, 0), Enum.EasingDirection.Out, Enum.EasingStyle.Quad, 0.15, true)
    end
    if progressText and text then
        progressText.Text = text
    end
end

local function currentKey()
    return trim((env and (rawget(env, "script_key") or rawget(env, "Key") or rawget(env, "KEY"))) or rawget(_G, "script_key") or rawget(_G, "Key") or rawget(_G, "KEY"))
end

local function saveKey(key)
    key = trim(key)
    if key == "" then return false end

    if env then
        env.script_key = key
        env.Key = key
        env.KEY = key
    end

    script_key = key
    _G.script_key = key
    _G.Key = key
    _G.KEY = key
    return true
end

local function saveState(tier, key)
    local stand = rawget(env or {}, "STAND")
    if type(stand) ~= "table" then stand = {} end
    stand.Config = type(stand.Config) == "table" and stand.Config or {}
    stand.State = type(stand.State) == "table" and stand.State or {}
    stand.UI = type(stand.UI) == "table" and stand.UI or {}

    local main = trim((env and (rawget(env, "MainUsername") or rawget(env, "MAIN_USERNAME"))) or rawget(_G, "MainUsername") or rawget(_G, "MAIN_USERNAME"))

    stand.Config.MainUsername = main
    stand.Config.mainUsername = main
    stand.Config.DisableInternalKeySystem = true
    stand.Config.KeySystemEnabled = false
    stand.Config.SecurityProvider = "polsec"
    stand.Config.PolsecEnabled = true
    stand.Config.RequirePolsecIntegrity = false
    stand.Config.KeyTier = tier
    stand.Config.Key = key
    stand.Config.key = key
    stand.Config.FreeKeyUrl = cfg.getKey

    stand.State.MainUsername = main
    stand.State.CommandOwnerName = main
    stand.State.SecurityProvider = "polsec"
    stand.State.PolsecEnabled = true
    stand.State.KeyTier = tier
    stand.State.PremiumAccess = tier == "premium"

    env.STAND = stand
    env.STAND_LOADER_CONFIG = stand.Config
    env.STAND_SECURITY_PROVIDER = "polsec"
    env.STAND_KEY_TIER = tier
    env.STAND_PREMIUM_ACCESS = tier == "premium"
    env.MainUsername = main
    env.MAIN_USERNAME = main

    _G.STAND = stand
    _G.STAND_LOADER_CONFIG = stand.Config
    _G.STAND_SECURITY_PROVIDER = "polsec"
    _G.STAND_KEY_TIER = tier
    _G.STAND_PREMIUM_ACCESS = tier == "premium"
    _G.MainUsername = main
    _G.MAIN_USERNAME = main
end

local function copy(text)
    local ok = false
    pcall(function()
        if setclipboard then
            setclipboard(text)
            ok = true
        end
    end)
    return ok
end

local function runScript(tier, status)
    if busy then return end

    local key = trim(guiRef and guiRef:FindFirstChild("KeyInput", true) and guiRef:FindFirstChild("KeyInput", true).Text or currentKey())
    if key == "" then
        status.Text = "Paste your key first."
        status.TextColor3 = Color3.fromRGB(255, 195, 105)
        setProgress(0, "Waiting")
        return
    end

    local url = tier == "premium" and cfg.premiumScript or cfg.freeScript
    busy = true
    saveKey(key)
    saveState(tier, key)

    status.Text = tier == "premium" and "Loading premium..." or "Loading free..."
    status.TextColor3 = Color3.fromRGB(145, 220, 255)
    setProgress(0.2, "Connecting")

    task.spawn(function()
        setProgress(0.42, "Downloading")
        local ok, body = pcall(function()
            return game:HttpGet(url)
        end)

        if not ok then
            busy = false
            status.Text = "Could not load Coroutine."
            status.TextColor3 = Color3.fromRGB(255, 120, 120)
            setProgress(0, "Failed")
            warn("[Coroutine] " .. tostring(body))
            return
        end

        body = tostring(body or "")
        if body == "" then
            busy = false
            status.Text = "Empty response."
            status.TextColor3 = Color3.fromRGB(255, 120, 120)
            setProgress(0, "Failed")
            return
        end

        setProgress(0.7, "Checking")
        local fn, err = loadstring(body)
        if not fn then
            busy = false
            status.Text = "Load failed."
            status.TextColor3 = Color3.fromRGB(255, 120, 120)
            setProgress(0, "Failed")
            warn("[Coroutine] " .. tostring(err))
            return
        end

        status.Text = "Starting..."
        setProgress(1, "Loaded")
        task.wait(0.12)
        cleanup()

        local runOk, runErr = xpcall(fn, function(e)
            return tostring(e) .. "\n" .. tostring(debug and debug.traceback and debug.traceback() or "")
        end)
        if not runOk then
            warn("[Coroutine] " .. tostring(runErr))
        end
    end)
end

-- COROUTINE STAND KEY PRELOADER

-- loader_polsec_only.txt from your hosted raw URL.
-- ============================================================

local PRELOADER_CONFIG = {
    LoaderUrl = "https://raw.githubusercontent.com/MunkeyHacker/ecfjxzsycgi/refs/heads/main/loader_polsec_only.txt",
    FreeKeyUrl = "https://getpolsec.com/ad/2cd45a4bf7f0a42e3abaa4db63d64b9bf397340010771b0f1257f30be89f1acb",
    Title = "Coroutine Stand",
}

if not game:IsLoaded() then
    pcall(function()
        game.Loaded:Wait()
    end)
end

local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
while not LocalPlayer do
    Players:GetPropertyChangedSignal("LocalPlayer"):Wait()
    LocalPlayer = Players.LocalPlayer
end

local function trim(value)
    value = tostring(value or "")
    return value:match("^%s*(.-)%s*$") or ""
end

local function isPlaceholder(value)
    value = trim(value)
    return value == "" or value:find("PASTE_", 1, true) ~= nil or value:find("YOUR_", 1, true) ~= nil
end

local function getEnvironment()
    if getgenv then
        local ok, env = pcall(getgenv)
        if ok and type(env) == "table" then
            return env
        end
    end
    return _G
end

local ENV = getEnvironment()
local ACTIVE_GUI = nil

local STAND = rawget(ENV or {}, "STAND")
if type(STAND) ~= "table" then
    STAND = {}
end
STAND.UI = type(STAND.UI) == "table" and STAND.UI or {}
STAND.UI.KeyPreloader = type(STAND.UI.KeyPreloader) == "table" and STAND.UI.KeyPreloader or {}
ENV.STAND = STAND
_G.STAND = STAND

local connections = {}

local function disconnect(connection)
    if connection and typeof(connection) == "RBXScriptConnection" then
        pcall(function()
            connection:Disconnect()
        end)
    end
end

local function getUiParents()
    local parents = {}

    local ok, hui = pcall(function()
        return gethui and gethui() or nil
    end)
    if ok and hui then
        table.insert(parents, hui)
    end

    local playerGui = LocalPlayer:FindFirstChildOfClass("PlayerGui") or LocalPlayer:WaitForChild("PlayerGui", 10)
    if playerGui then
        table.insert(parents, playerGui)
    end

    local coreGui
    ok, coreGui = pcall(function()
        return game:GetService("CoreGui")
    end)
    if ok and coreGui then
        table.insert(parents, coreGui)
    end

    return parents
end

local function cleanup()
    for _, connection in ipairs(connections) do
        disconnect(connection)
    end
    connections = {}

    pcall(function()
        if ACTIVE_GUI and ACTIVE_GUI.Parent then
            ACTIVE_GUI:Destroy()
        end
    end)
    ACTIVE_GUI = nil

    for _, parent in ipairs(getUiParents()) do
        pcall(function()
            local oldGui = parent:FindFirstChild("CoroutinePolsecKeyPreloader")
            if oldGui then oldGui:Destroy() end
            oldGui = parent:FindFirstChild("StandPolsecKeyPreloader")
            if oldGui then oldGui:Destroy() end
        end)
    end
end

if type(STAND.UI.KeyPreloader.Cleanup) == "function" then
    pcall(STAND.UI.KeyPreloader.Cleanup)
end
STAND.UI.KeyPreloader.Cleanup = cleanup

local function track(connection)
    if connection and typeof(connection) == "RBXScriptConnection" then
        table.insert(connections, connection)
    end
    return connection
end

local function firstUiParent()
    local parents = getUiParents()
    return parents[1]
end

local function makeCorner(parent, radius)
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, radius or 8)
    corner.Parent = parent
    return corner
end

local function makeStroke(parent, color, transparency)
    local stroke = Instance.new("UIStroke")
    stroke.Color = color or Color3.fromRGB(76, 92, 130)
    stroke.Transparency = transparency or 0.2
    stroke.Thickness = 1
    stroke.Parent = parent
    return stroke
end

local function currentKey()
    local envKey = ENV and (rawget(ENV, "script_key") or rawget(ENV, "Key") or rawget(ENV, "KEY"))
    local globalKey = rawget(_G, "script_key") or rawget(_G, "Key") or rawget(_G, "KEY")
    return trim(envKey or globalKey or "")
end

local function publishKey(key)
    key = trim(key)
    if key == "" then
        return false
    end

    if ENV then
        ENV.script_key = key
        ENV.Key = key
        ENV.KEY = key
    end
    pcall(function()
        script_key = key
    end)
    _G.script_key = key
    _G.Key = key
    _G.KEY = key
    return true
end

local function copyText(value)
    local ok = false
    pcall(function()
        if setclipboard then
            setclipboard(value)
            ok = true
        end
    end)
    return ok
end

local progressFill = nil
local progressText = nil
local loadInProgress = false

local function setProgress(value, text)
    value = math.clamp(tonumber(value) or 0, 0, 1)
    if progressFill then
        progressFill:TweenSize(UDim2.new(value, 0, 1, 0), Enum.EasingDirection.Out, Enum.EasingStyle.Quad, 0.15, true)
    end
    if progressText and text then
        progressText.Text = text
    end
end

local function loadPolsecOnlyLoader(statusLabel)
    if loadInProgress then
        return
    end

    local loaderUrl = trim(PRELOADER_CONFIG.LoaderUrl)
    if isPlaceholder(loaderUrl) then
        statusLabel.Text = "Upload loader_polsec_only.txt and paste its raw URL into LoaderUrl."
        statusLabel.TextColor3 = Color3.fromRGB(255, 195, 105)
        setProgress(0, "Waiting")
        return
    end

    loadInProgress = true
    setProgress(0.12, "Preparing")
    statusLabel.Text = "Loading..."
    statusLabel.TextColor3 = Color3.fromRGB(145, 220, 255)

    task.spawn(function()
        setProgress(0.3, "Downloading")
        local ok, sourceOrError = pcall(function()
            return game:HttpGet(loaderUrl)
        end)
        if not ok then
            loadInProgress = false
            setProgress(0, "Failed")
            statusLabel.Text = "Could not download loader."
            statusLabel.TextColor3 = Color3.fromRGB(255, 120, 120)
            warn("[Preloader] " .. tostring(sourceOrError))
            return
        end

        local source = tostring(sourceOrError or "")
        if source == "" then
            loadInProgress = false
            setProgress(0, "Failed")
            statusLabel.Text = "Loader download was empty."
            statusLabel.TextColor3 = Color3.fromRGB(255, 120, 120)
            return
        end

        setProgress(0.62, "Checking")
        local chunk, compileError = loadstring(source)
        if not chunk then
            loadInProgress = false
            setProgress(0, "Failed")
            statusLabel.Text = "Loader compile failed."
            statusLabel.TextColor3 = Color3.fromRGB(255, 120, 120)
            warn("[Preloader] " .. tostring(compileError))
            return
        end

        setProgress(0.88, "Starting")
        statusLabel.Text = "Starting loader..."
        task.wait(0.2)
        setProgress(1, "Loaded")
        task.wait(0.08)
        cleanup()

        local runOk, runError = xpcall(chunk, function(err)
            return tostring(err) .. "\n" .. tostring(debug and debug.traceback and debug.traceback() or "")
        end)
        if not runOk then
            warn("[Preloader] " .. tostring(runError))
            return
        end
    end)
end

local LOADER_CONFIG = {
    MainUsername = "", -- Optional: set this to skip the player selector.
    PolsecScriptUrl = "https://api.getpolsec.com/scripts/hosted/2cd45a4bf7f0a42e3abaa4db63d64b9bf397340010771b0f1257f30be89f1acb.lua",
    -- Public users should set _G.Key or _G.script_key before running this loader.
    -- Free keys come from this Polsec/LootLabs page.
    PolsecKey = "",
    FreeKeyUrl = "https://getpolsec.com/ad/2cd45a4bf7f0a42e3abaa4db63d64b9bf397340010771b0f1257f30be89f1acb",
    PremiumKeyCheckUrl = "https://raw.githubusercontent.com/MunkeyHacker/dddddddssscccXXJk_s/refs/heads/main/ke%20ys",
    RequirePolsecIntegrity = true,
    RefreshInterval = 2,
}

if not game:IsLoaded() then
    pcall(function()
        game.Loaded:Wait()
    end)
end

local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
while not LocalPlayer do
    Players:GetPropertyChangedSignal("LocalPlayer"):Wait()
    LocalPlayer = Players.LocalPlayer
end

local function trim(value)
    value = tostring(value or "")
    return value:match("^%s*(.-)%s*$") or ""
end

local function isPlaceholder(value)
    value = trim(value)
    return value == "" or value:find("PASTE_", 1, true) ~= nil or value:find("YOUR_", 1, true) ~= nil
end

local function getEnvironment()
    if getgenv then
        local ok, env = pcall(getgenv)
        if ok and type(env) == "table" then
            return env
        end
    end
    return _G
end

local STAND_ENV = getEnvironment()

local function getUiParents()
    local parents = {}

    local ok, hui = pcall(function()
        return gethui and gethui() or nil
    end)
    if ok and hui then
        table.insert(parents, hui)
    end

    local playerGui = LocalPlayer:FindFirstChildOfClass("PlayerGui") or LocalPlayer:WaitForChild("PlayerGui", 10)
    if playerGui then
        table.insert(parents, playerGui)
    end

    local coreGui
    ok, coreGui = pcall(function()
        return game:GetService("CoreGui")
    end)
    if ok and coreGui then
        table.insert(parents, coreGui)
    end

    return parents
end

local function firstUiParent()
    local parents = getUiParents()
    return parents[1]
end

local STAND = rawget(STAND_ENV or {}, "STAND")
if type(STAND) ~= "table" then
    STAND = {}
end
STAND.Config = type(STAND.Config) == "table" and STAND.Config or {}
STAND.State = type(STAND.State) == "table" and STAND.State or {}
STAND.UI = type(STAND.UI) == "table" and STAND.UI or {}
STAND.UI.PolsecLoader = type(STAND.UI.PolsecLoader) == "table" and STAND.UI.PolsecLoader or {}

STAND_ENV.STAND = STAND
_G.STAND = STAND

if type(STAND.UI.PolsecLoader.Cleanup) == "function" then
    pcall(STAND.UI.PolsecLoader.Cleanup)
end

local loaderConnections = {}
STAND.UI.PolsecLoader.Connections = loaderConnections
STAND.State.PolsecLoaderRunToken = tostring(os.clock()) .. ":" .. tostring(math.random(100000, 999999))
local LOADER_RUN_TOKEN = STAND.State.PolsecLoaderRunToken

local function isCurrentLoaderRun()
    return STAND.State.PolsecLoaderRunToken == LOADER_RUN_TOKEN
end

local function disconnectConnection(connection)
    if connection and typeof(connection) == "RBXScriptConnection" then
        pcall(function()
            connection:Disconnect()
        end)
    end
end

local function cleanupLoaderUi()
    for _, connection in ipairs(loaderConnections) do
        disconnectConnection(connection)
    end
    loaderConnections = {}
    STAND.UI.PolsecLoader.Connections = loaderConnections

    for _, parent in ipairs(getUiParents()) do
        pcall(function()
            local selector = parent:FindFirstChild("StandPolsecMainSelector")
            if selector then selector:Destroy() end
            local statusGui = parent:FindFirstChild("StandPolsecLoaderStatus")
            if statusGui then statusGui:Destroy() end
        end)
    end
end

STAND.UI.PolsecLoader.Cleanup = cleanupLoaderUi

local function trackConnection(connection)
    if connection and typeof(connection) == "RBXScriptConnection" then
        table.insert(loaderConnections, connection)
    end
    return connection
end

local function makeCorner(parent, radius)
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, radius or 8)
    corner.Parent = parent
    return corner
end

local function makeStroke(parent, color, transparency)
    local stroke = Instance.new("UIStroke")
    stroke.Color = color or Color3.fromRGB(80, 90, 120)
    stroke.Transparency = transparency or 0.3
    stroke.Thickness = 1
    stroke.Parent = parent
    return stroke
end

local function removeOldSelector()
    for _, parent in ipairs(getUiParents()) do
        pcall(function()
            local oldGui = parent:FindFirstChild("StandPolsecMainSelector")
            if oldGui then oldGui:Destroy() end
        end)
    end
end

local function configuredMainUsername()
    local envName = STAND_ENV and (rawget(STAND_ENV, "MainUsername") or rawget(STAND_ENV, "MAIN_USERNAME"))
    local globalName = rawget(_G, "MainUsername") or rawget(_G, "MAIN_USERNAME")
    return trim(envName or globalName or LOADER_CONFIG.MainUsername or "")
end

local function getFreshPlayers()
    local result = {}
    for _, player in ipairs(Players:GetPlayers()) do
        if player and player.Parent == Players then
            table.insert(result, player)
        end
    end
    table.sort(result, function(a, b)
        return tostring(a.Name):lower() < tostring(b.Name):lower()
    end)
    return result
end

        if not isCurrentLoaderRun() or not gui.Parent then

    task.spawn(function()
        local interval = tonumber(LOADER_CONFIG.RefreshInterval) or 2
        while isCurrentLoaderRun() and gui.Parent and not selectedMainUsername do
            task.wait(interval)
            refreshList()
        end
    end)

    repeat task.wait() until selectedMainUsername or not gui.Parent or not isCurrentLoaderRun()
    return trim(selectedMainUsername or "")
end

local function normalizeKey(value)
    return trim(value):gsub("%s+", ""):lower()
end

local function resolvePolsecKey()
    local envKey = STAND_ENV and (rawget(STAND_ENV, "script_key") or rawget(STAND_ENV, "Key") or rawget(STAND_ENV, "KEY"))
    local globalKey = rawget(_G, "script_key") or rawget(_G, "Key") or rawget(_G, "KEY")
    return trim(envKey or globalKey or LOADER_CONFIG.PolsecKey or "")
end

local function fetchText(url)
    url = trim(url)
    if isPlaceholder(url) then
        return nil
    end

    local ok, body = pcall(function()
        return game:HttpGet(url)
    end)
    if ok and type(body) == "string" and body ~= "" then
        return body
    end
    return nil
end

local function responseContainsKey(key, response)
    local wanted = normalizeKey(key)
    if wanted == "" then
        return false
    end

    response = tostring(response or "")
    for line in response:gmatch("[^\r\n]+") do
        local cleanLine = line:gsub("%-%-.*$", "")
        for entry in cleanLine:gmatch("[^,;%s]+") do
            if normalizeKey(entry) == wanted then
                return true
            end
        end
        for token in cleanLine:gmatch("[%w_%-]+") do
            if normalizeKey(token) == wanted then
                return true
            end
        end
    end
    return normalizeKey(response) == wanted
end

local function resolvePolsecTier(key)
    local premiumBody = fetchText(LOADER_CONFIG.PremiumKeyCheckUrl)
    if premiumBody and responseContainsKey(key, premiumBody) then
        return "premium"
    end
    return "free"
end

local function publishPolsecKey(key)
    key = trim(key)
    if key == "" then
        return false
    end

    if STAND_ENV then
        STAND_ENV.script_key = key
        STAND_ENV.Key = key
        STAND_ENV.KEY = key
    end
    pcall(function()
        script_key = key
    end)
    _G.script_key = key
    _G.Key = key
    _G.KEY = key
    return true
end

local function publishConfig(mainUsername, keyTier, activeKey)
    local cleanName = trim(mainUsername or "")
    keyTier = tostring(keyTier or "free"):lower()
    activeKey = trim(activeKey or "")

    STAND.Config.MainUsername = cleanName
    STAND.Config.mainUsername = cleanName
    STAND.Config.DisableInternalKeySystem = true
    STAND.Config.KeySystemEnabled = false
    STAND.Config.SecurityProvider = "polsec"
    STAND.Config.PolsecEnabled = true
    STAND.Config.RequirePolsecIntegrity = LOADER_CONFIG.RequirePolsecIntegrity == true
    STAND.Config.KeyTier = keyTier
    STAND.Config.Key = activeKey
    STAND.Config.key = activeKey
    STAND.Config.FreeKeyUrl = LOADER_CONFIG.FreeKeyUrl

    STAND.State.MainUsername = cleanName
    STAND.State.CommandOwnerName = cleanName
    STAND.State.LoaderSelectedAt = os.time()
    STAND.State.SecurityProvider = "polsec"
    STAND.State.PolsecEnabled = true
    STAND.State.KeyTier = keyTier
    STAND.State.PremiumAccess = keyTier == "premium"

    STAND.UI.PolsecLoader.SelectedMainUsername = cleanName
    STAND.UI.PolsecLoader.KeyTier = keyTier

    STAND_ENV.STAND = STAND
    STAND_ENV.STAND_LOADER_CONFIG = STAND.Config
    STAND_ENV.MAIN_USERNAME = cleanName
    STAND_ENV.MainUsername = cleanName
    STAND_ENV.STAND_SECURITY_PROVIDER = "polsec"
    STAND_ENV.STAND_KEY_TIER = keyTier
    STAND_ENV.STAND_PREMIUM_ACCESS = keyTier == "premium"

    _G.STAND = STAND
    _G.STAND_LOADER_CONFIG = STAND.Config
    _G.MAIN_USERNAME = cleanName
    _G.MainUsername = cleanName
    _G.STAND_SECURITY_PROVIDER = "polsec"
    _G.STAND_KEY_TIER = keyTier
    _G.STAND_PREMIUM_ACCESS = keyTier == "premium"
end

local function fetchPolsecScript()
    local url = trim(LOADER_CONFIG.PolsecScriptUrl)
    if isPlaceholder(url) then
        return nil, "PolsecScriptUrl is not set"
    end

    local ok, body = pcall(function()
        return game:HttpGet(url)
    end)
    if not ok then
        return nil, tostring(body)
    end
    if type(body) ~= "string" or body == "" then
        return nil, "Polsec returned an empty script"
    end
    return body, nil
end

local function runPolsecScript(source)
    local chunk, compileError = loadstring(source)
    if not chunk then
        return false, compileError
    end

    local ok, runtimeError = xpcall(chunk, function(err)
        return tostring(err) .. "\n" .. tostring(debug and debug.traceback and debug.traceback() or "")
    end)
    if not ok then
        return false, runtimeError
    end
    return true, nil
end

local mainUsername = waitForMainUserSelection()
if not isCurrentLoaderRun() then return end
if mainUsername == "" then
    warn("[PolsecLoader] No main user selected")
    return
end

local activeKey = resolvePolsecKey()
local keyTier = "free"
if activeKey ~= "" then
    publishPolsecKey(activeKey)
    keyTier = resolvePolsecTier(activeKey)
end

publishConfig(mainUsername, keyTier, activeKey)

local statusText = "Selected " .. tostring(mainUsername) .. " - loading Polsec script"
if activeKey ~= "" then
    statusText = statusText .. " (" .. tostring(keyTier) .. ")"
else
    statusText = statusText .. " (Polsec will ask for key)"
end
local _, statusLabel = showLoaderStatus(statusText .. "...", false)
local source, fetchError = fetchPolsecScript()
if not source then
    local message = "Loader failed: " .. tostring(fetchError)
    warn("[PolsecLoader] " .. message)
    showLoaderStatus(message, true)
    return
end

if not isCurrentLoaderRun() then return end
if statusLabel then
    statusLabel.Text = "Polsec script downloaded - starting..."
end

local ok, runError = runPolsecScript(source)
if not ok then
    local message = "Loader failed: " .. tostring(runError)
    warn("[PolsecLoader] " .. message)
    showLoaderStatus(message, true)
    return
end

if statusLabel then
    statusLabel.Text = "Loaded Polsec script (" .. tostring(keyTier) .. ")"
    task.delay(2, function()
        pcall(function()
            local gui = statusLabel:FindFirstAncestorOfClass("ScreenGui")
            if gui then gui:Destroy() end
        end)
    end)
end

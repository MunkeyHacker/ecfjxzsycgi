local function urlEncode(value)
    local ok, encoded = pcall(function()
        return HttpService:UrlEncode(tostring(value or ""))
    end)
    if ok and type(encoded) == "string" then
        return encoded
    end
    return tostring(value or ""):gsub("([^%w%-_%.~])", function(c)
        return string.format("%%%02X", string.byte(c))
    end)
end

local function currentHwid()
    local candidates = {}

    if type(gethwid) == "function" then
        local ok, value = pcall(gethwid)
        if ok then
            candidates[#candidates + 1] = value
        end
    end

    if env then
        candidates[#candidates + 1] = rawget(env, "HWID")
        candidates[#candidates + 1] = rawget(env, "hwid")
    end

    local ok, analytics = pcall(function()
        return game:GetService("RbxAnalyticsService")
    end)
    if ok and analytics then
        ok, analytics = pcall(function()
            return analytics:GetClientId()
        end)
        if ok then
            candidates[#candidates + 1] = analytics
        end
    end

    if LocalPlayer then
        candidates[#candidates + 1] = tostring(LocalPlayer.UserId)
    end

    for _, value in ipairs(candidates) do
        value = trim(value)
        if value ~= "" then
            return value
        end
    end

    return "unknown"
end

local function keyMark(key)
    key = normalizeKey(key)
    local h1, h2, h3 = 17, 29, 43

    for i = 1, #key do
        local b = string.byte(key, i)
        h1 = (h1 * 131 + b) % 1000000007
        h2 = (h2 * 137 + b) % 1000000009
        h3 = (h3 * 149 + b) % 1000000033
    end

    return tostring(h1) .. ":" .. tostring(h2) .. ":" .. tostring(h3)
end

local PREMIUM_DURATIONS = {
    ["1h"] = 60 * 60,
    ["24h"] = 24 * 60 * 60,
    ["1d"] = 24 * 60 * 60,
    ["2d"] = 2 * 24 * 60 * 60,
    ["1m"] = 30 * 24 * 60 * 60,
    ["3m"] = 90 * 24 * 60 * 60,
    ["1y"] = 365 * 24 * 60 * 60,
    ["10y"] = 10 * 365 * 24 * 60 * 60,
}

local function premiumDurationSeconds(value)
    value = trim(value):lower()
    if value == "never" or value == "permanent" or value == "perm" then
        return nil, true
    end
    if PREMIUM_DURATIONS[value] then
        return PREMIUM_DURATIONS[value], false
    end

    local amount, unit = value:match("^(%d+%.?%d*)%s*([%a]+)$")
    amount = tonumber(amount)
    if not amount or amount <= 0 then
        return nil, false
    end

    if unit == "minute" or unit == "minutes" or unit == "min" or unit == "mins" then
        return amount * 60, false
    end
    if unit == "hour" or unit == "hours" or unit == "hr" or unit == "hrs" or unit == "h" then
        return amount * 60 * 60, false
    end
    if unit == "day" or unit == "days" or unit == "d" then
        return amount * 24 * 60 * 60, false
    end
    if unit == "week" or unit == "weeks" or unit == "w" then
        return amount * 7 * 24 * 60 * 60, false
    end
    if unit == "month" or unit == "months" or unit == "mo" or unit == "mos" then
        return amount * 30 * 24 * 60 * 60, false
    end
    if unit == "year" or unit == "years" or unit == "yr" or unit == "yrs" or unit == "y" then
        return amount * 365 * 24 * 60 * 60, false
    end

    return nil, false
end

local function downloadText(url)

        return game:HttpGet(url)
    end)

local function looksLikePolsecReject(body)
    body = tostring(body or ""):lower()
    if body == "" then
        return false
    end

    if body:find(":kick%(") or body:find("%.kick%(") then
        return true
    end
    if body:find("invalid key", 1, true) or body:find("incorrect key", 1, true) then
        return true
    end
    if body:find("key expired", 1, true) or body:find("blacklisted", 1, true) then
        return true
    end
    if body:find("hwid", 1, true) and body:find("mismatch", 1, true) then
        return true
    end

    return false
end

local function parsePlainPremiumList(body)
    local list = { keys = {} }
    for line in tostring(body or ""):gmatch("[^\r\n]+") do
        line = trim(line)
        if line ~= "" and line:sub(1, 1) ~= "#" and line:sub(1, 2) ~= "--" then
            local key, duration, issuedAt = line:match("^([^,]+),([^,]+),?(.*)$")
            if not key then
                key, duration, issuedAt = line:match("^([^|]+)|([^|]+)|?(.*)$")
            end
            key = trim(key or line)
            if key ~= "" then
                duration = trim(duration or "never")
                issuedAt = tonumber(trim(issuedAt or ""))
                list.keys[key] = { duration = duration ~= "" and duration or "never", issuedAt = issuedAt }
            end
        end
    end
    return list
end

local function mergePremiumLists(into, from)
    into.keys = type(into.keys) == "table" and into.keys or {}
    if type(from) ~= "table" then
        return into
    end

    local maps = { from.keys, from.premium }
    for _, map in ipairs(maps) do
        if type(map) == "table" then
            for key, entry in pairs(map) do
                into.keys[key] = entry
            end
        end
    end

    if type(from.fingerprints) == "table" then
        into.fingerprints = type(into.fingerprints) == "table" and into.fingerprints or {}
        for mark, entry in pairs(from.fingerprints) do
            into.fingerprints[mark] = entry
        end
    end

    if not into.issuedAt then
        into.issuedAt = from.issuedAt or from.generatedAt
    end

    return into
end

local function parseIssuedAtCall(body)
    body = tostring(body or "")
    local args = body:match("DEFAULT_ISSUED_AT%s*=%s*issuedAt%s*%((.-)%)")
    if not args then
        return nil
    end

    local values = {}
    for number in args:gmatch("%d+") do
        values[#values + 1] = tonumber(number)
    end

    if not values[1] or not values[2] or not values[3] then
        return nil
    end

    return os.time({
        year = values[1],
        month = values[2],
        day = values[3],
        hour = values[4] or 0,
        min = values[5] or 0,
        sec = values[6] or 0,
    })
end

local function parsePremiumKeyLinesBlock(body)
    local list = { keys = {} }
    body = tostring(body or "")

    for block in body:gmatch("PREMIUM_KEY_LINES%s*=%s*%[%[(.-)%]%]") do
        mergePremiumLists(list, parsePlainPremiumList(block))
    end

    list.issuedAt = parseIssuedAtCall(body)

    return list
end

local function decodeBytePayload(body)
    body = tostring(body or "")
    local payload = body:match("local%s+b%s*=%s*%{(.-)%}") or body:match("b%s*=%s*%{(.-)%}")
    if not payload then
        return nil
    end

    local chars = {}
    for number in payload:gmatch("%d+") do
        number = tonumber(number)
        if number and number >= 0 and number <= 255 then
            chars[#chars + 1] = string.char(number)
        end
    end

    if #chars == 0 then
        return nil
    end

    return table.concat(chars)
end

local function parseTableStylePremiumKeys(body)
    local list = { keys = {} }
    body = tostring(body or "")

    for key, entryBody in body:gmatch("%[\"([^\"]+)\"%]%s*=%s*%{(.-)%}") do
        local duration = entryBody:match("duration%s*=%s*\"([^\"]+)\"") or entryBody:match("expires%s*=%s*\"([^\"]+)\"") or "never"
        local issuedAt = tonumber(entryBody:match("issuedAt%s*=%s*(%d+)") or "")
        local expiresAt = tonumber(entryBody:match("expiresAt%s*=%s*(%d+)") or "")
        list.keys[key] = { duration = duration, issuedAt = issuedAt, expiresAt = expiresAt }
    end

    return list
end

local function parseExecutablePremiumList(body)
    local list = { keys = {} }
    body = tostring(body or "")
    if body == "" or looksLikePolsecReject(body) then
        return list
    end

    local loader = loadstring or load
    if type(loader) ~= "function" then
        return list
    end

    local fn = loader(body)
    if type(fn) ~= "function" then
        return list
    end

    local ok, result = pcall(fn)
    if ok and type(result) == "table" then
        mergePremiumLists(list, result)
    end

    return list
end

local function decodePremiumList(body, allowExecution)
    local list = { keys = {} }
    local decoded = decodeBytePayload(body)
    if decoded and decoded ~= body then
        mergePremiumLists(list, decodePremiumList(decoded, allowExecution))
    end

    mergePremiumLists(list, parsePremiumKeyLinesBlock(body))
    mergePremiumLists(list, parseTableStylePremiumKeys(body))
    mergePremiumLists(list, parsePlainPremiumList(body))
    if allowExecution == true then
        mergePremiumLists(list, parseExecutablePremiumList(body))
    end

    return list, nil
end

local function validatePolsecKey(publicKey, key)
    publicKey = trim(publicKey)
    if publicKey == "" then
        return true, "missing-public-key"
    end

    local url = "https://api.getpolsec.com/external?public_key=" .. urlEncode(publicKey)
        .. "&key=" .. urlEncode(key)
        .. "&hwid=" .. urlEncode(currentHwid())

    local ok, body = pcall(function()
        return game:HttpGet(url)
    end)
    if not ok then
        return false, tostring(body)
    end

    body = tostring(body or "")
    local decodedOk, decoded = pcall(function()
        return HttpService:JSONDecode(body)
    end)
    if decodedOk and type(decoded) == "table" then
        return decoded.valid == true, tostring(decoded.message or "")
    end

    return body:lower():find('"valid"%s*:%s*true') ~= nil, body
end

local function validateMergedKey(key)
    return validatePolsecKey(mergedPublicKey(), key)
end

local function validatePremiumKey(key)
    return validatePolsecKey(premiumPublicKey(), key)
end

local function loadPremiumList()
    local urls = premiumListUrls()
    if #urls == 0 then
        return nil, "premium list not configured"
    end

    local merged = { keys = {}, fingerprints = {} }
    local lastErr
    local loaded = false

    for _, url in ipairs(urls) do
        local body, err = downloadText(url)
        if body and not looksLikePolsecReject(body) then
            local list, parseErr = decodePremiumList(body, cfg.executePremiumListWrapper == true)
            if type(list) == "table" then
                mergePremiumLists(merged, list)
                loaded = true
            else
                lastErr = parseErr or "premium list did not return a table"
            end
        else
            lastErr = err or "premium list rejected key"
        end
    end

    if not loaded then
        return nil, lastErr or "premium list unavailable"
    end

    STAND.Cache.PremiumKeyList = merged
    STAND.Cache.PremiumKeyListCheckedAt = os.time()
    return merged, nil
end

local function premiumEntryActive(entry, list)
    local now = os.time()
    if entry == true then
        return true, { duration = "never" }
    end
    if type(entry) == "string" then
        entry = { duration = entry }
    end
    if type(entry) ~= "table" then
        return false, { reason = "bad-entry" }
    end
    if entry.active == false or entry.enabled == false then
        return false, { reason = "disabled" }
    end

    local duration = trim(entry.duration or entry.expires or entry.expiration or entry[1] or "never"):lower()
    local expiresAt = tonumber(entry.expiresAt or entry.expiresAtUnix or entry.expireAt)
    if expiresAt then
        if now <= expiresAt then
            return true, { duration = duration, expiresAt = expiresAt }
        end
        return false, { reason = "expired", duration = duration, expiresAt = expiresAt }
    end

    local seconds, never = premiumDurationSeconds(duration)
    if never then
        return true, { duration = "never" }
    end
    if not seconds then
        return false, { reason = "bad-duration", duration = duration }
    end

    local issuedAt = tonumber(entry.issuedAt or entry.createdAt or entry.startsAt or (list and (list.issuedAt or list.generatedAt)))
    if not issuedAt then
        return false, { reason = "missing-issuedAt", duration = duration }
    end

    expiresAt = issuedAt + seconds
    if now <= expiresAt then
        return true, { duration = duration, issuedAt = issuedAt, expiresAt = expiresAt }
    end
    return false, { reason = "expired", duration = duration, issuedAt = issuedAt, expiresAt = expiresAt }
end

local function findPremiumEntry(list, key)
    local wanted = normalizeKey(key)
    local mark = keyMark(key)
    local maps = { list and list.keys, list and list.premium, list and list.fingerprints }
    for _, map in ipairs(maps) do
        if type(map) == "table" then
            if map[wanted] ~= nil then
                return map[wanted], wanted
            end
            if map[mark] ~= nil then
                return map[mark], mark
            end
            for listedKey, entry in pairs(map) do
                if normalizeKey(listedKey) == wanted or tostring(listedKey) == mark then
                    return entry, tostring(listedKey)
                end
            end
        end
    end
    return nil, nil
end

local function applyRuntimeKey(key)
    key = trim(key)
    if key == "" then return false end

    if env then
        env.script_key = key
        env.Key = key
        env.KEY = key
    end

    script_key = key
    _G.script_key = key
    _G.Key = key
    _G.KEY = key
    return true
end

local function checkPremiumKey(key)
    applyRuntimeKey(key)

    local premiumPublic = premiumPublicKey()
    local externalMessage
    if premiumPublic ~= "" then
        local valid, message = validatePremiumKey(key)
        if valid then
            return true, { duration = "never", reason = "premium-polsec", publicKey = premiumPublic }
        end
        externalMessage = message
    end

    local list, err = loadPremiumList()
    if not list then
        if premiumPublic ~= "" then
            return false, { reason = "not-premium", message = externalMessage or err, publicKey = premiumPublic }
        end
        return nil, { reason = "unavailable", message = err }
    end
    local entry, matchedKey = findPremiumEntry(list, key)
    if entry == nil then
        return false, { reason = "not-premium", message = externalMessage, publicKey = premiumPublic }
    end
    local active, meta = premiumEntryActive(entry, list)
    meta = type(meta) == "table" and meta or {}
    meta.matchedKey = matchedKey
    if active then
        return true, meta
    end
    return false, meta
end

local function keyTier(key, status)
    if status then
        status.Text = "Checking premium key..."
        status.TextColor3 = Color3.fromRGB(145, 220, 255)
    end

    local premium, meta = checkPremiumKey(key)
    if premium == true then
        return "premium", "premium-list", meta
    end

    meta = type(meta) == "table" and meta or {}
    meta.duration = meta.duration or "free"
    meta.premiumCheckReason = meta.reason or (premium == nil and "unavailable" or "not-premium")
    return "free", "merged-polsec", meta
end

local function saveState(tier, key, mode, reason, meta)
    mode = "auto"

    local tierValue = accessTier(tier)
    local stand = rawget(env or {}, "STAND")

local function copy(text)
    local ok = false
    pcall(function()
        if setclipboard then
            setclipboard(text)
            ok = true
        end
    end)
    return ok
end

local function download(url)
    local body, err = downloadText(url)
    if not body then
        return nil, err
    end

    if looksLikePolsecReject(body) then
        return nil, "invalid key"
    end

    local fn, err = loadstring(body)
    if not fn then
        return nil, tostring(err)
    end

    return fn, nil
end

local function startTier(tier, fn, key, mode, reason, meta)
    saveState(tier, key, mode, reason, meta)
    setProgress(1, tier == "premium" and "Premium" or tier == "free" and "Free" or "Key")
    task.wait(0.12)
    cleanup()

    local ok, err = xpcall(fn, function(e)
        return tostring(e) .. "\n" .. tostring(debug and debug.traceback and debug.traceback() or "")
    end)

    return ok, err
end

local function runScript(status)
    if busy then return end

    local key = trim(guiRef and guiRef:FindFirstChild("KeyInput", true) and guiRef:FindFirstChild("KeyInput", true).Text or currentKey())
    if key == "" then
        status.Text = "Paste your key first."
        status.TextColor3 = Color3.fromRGB(255, 195, 105)
        setProgress(0, "Waiting")
        return
    end

    local tierMode = "auto"
    busy = true
    local tier, tierReason, tierMeta = keyTier(key, status)
    local url = scriptUrlForTier(tier)

    local shouldValidateMerged = (tier == "premium" and cfg.requireMergedKeyForPremium == true) or (tier ~= "premium" and cfg.requireMergedKeyForFree ~= false)
    if shouldValidateMerged then
        status.Text = "Checking key..."
        status.TextColor3 = Color3.fromRGB(145, 220, 255)
        local valid, message = validateMergedKey(key)
        if not valid then
            busy = false
            status.Text = tier == "premium" and "Premium key missing in merged PolSec." or "Incorrect key."
            status.TextColor3 = Color3.fromRGB(255, 120, 120)
            setProgress(0, "Incorrect")
            warn("[Coroutine] " .. tostring(message or "invalid key"))
            return
        end
    end

    saveKey(key)
    saveState(tier, key, tierMode, tierReason, tierMeta)

    status.Text = tier == "premium" and "Premium accepted." or "Checking key..."
    status.TextColor3 = Color3.fromRGB(145, 220, 255)
    setProgress(0.2, "Connecting")

    task.spawn(function()
        setProgress(0.55, "Downloading")
        local fn, err = download(url)
        if not fn then
            busy = false
            status.Text = err == "invalid key" and (tier == "premium" and "Premium key not valid for premium script." or "Incorrect key.") or "Load failed."
            status.TextColor3 = Color3.fromRGB(255, 120, 120)
            setProgress(0, err == "invalid key" and "Incorrect" or "Failed")
            warn("[Coroutine] " .. tostring(err))
            return
        end

        local runOk, runErr = startTier(tier, fn, key, tierMode, tierReason, tierMeta)
        if not runOk then
            warn("[Coroutine] " .. tostring(runErr))
        end
    end)
end
